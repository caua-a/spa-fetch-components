/* Criar uma função geradora de erro */

function GeradorErros(mensagem, codigo, tipo){
    let erro = new Error(mensagem)
    erro.codigo = codigo
    erro.tipo = tipo

    return erro
}

const input_pix = document.querySelector('.input_pix')
const display_saldo = document.querySelector('.saldo')
const container_error = document.querySelector('.container')
const botao_enviar = document.querySelector('.botao_enviar')

const saldo = 200;
const pix = Number(input_pix.value)

const load_items = (saldo) =>{
    display_saldo.textContent = saldo
}

botao_enviar.addEventListener('click', (e)=>{
    try{

        if (saldo < pix){
            throw GeradorErros('Saldo Insuficiente',50,'FINANCEIRO')
        }

    } catch(erro){

        let text = `
        <span><strong>Nome do erro:</strong> ${erro.name}</span> <br />
        <span><strong>Mensagem:</strong> ${erro.message}</span> <br />
        <span><strong>Código:</strong> ${erro.codigo}</span> <br />
        <span><strong>Tipo:</strong> ${erro.tipo}</span> <br />
        <span><strong>Stack:</strong> ${erro.stack}</span>
        `;

        container_error.style.display='block';
        container_error.innerHTML = text;
        

    }
})

(window.onload = load_items(saldo))
